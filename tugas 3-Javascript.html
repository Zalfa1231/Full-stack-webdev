// Array produk awal let produkToko = [ { id: 1, nama: "Laptop", harga: 7000000, stok: 5 }, { id: 2, nama: "Mouse", harga: 200000, stok: 10 }, { id: 3, nama: "Keyboard", harga: 350000, stok: 7 } ]; // Fungsi untuk menampilkan produk function tampilkanProduk() { console.log("=== Daftar Produk Toko ==="); produkToko.forEach(p => { console.log(`ID: ${p.id}, Nama: ${p.nama}, Harga: Rp ${p.harga.toLocaleString()}, Stok: ${p.stok}`); }); console.log("==========================\n"); } // Fungsi untuk menambahkan produk baru function tambahProduk(nama, harga, stok) { // buat id baru dengan ambil id terakhir + 1 let idBaru = produkToko.length > 0 ? produkToko[produkToko.length - 1].id + 1 : 1; let produkBaru = { id: idBaru, nama: nama, harga: harga, stok: stok }; produkToko.push(produkBaru); console.log(`Produk "${nama}" berhasil ditambahkan!\n`); } // Fungsi untuk menghapus produk berdasarkan id function hapusProduk(id) { let index = produkToko.findIndex(p => p.id === id); if (index !== -1) { let hapus = produkToko.splice(index, 1); console.log(`Produk "${hapus[0].nama}" berhasil dihapus!\n`); } else { console.log(`Produk dengan ID ${id} tidak ditemukan!\n`); } } // Contoh penggunaan tampilkanProduk(); tambahProduk("Headset", 500000, 8); tampilkanProduk(); hapusProduk(2); tampilkanProduk();
